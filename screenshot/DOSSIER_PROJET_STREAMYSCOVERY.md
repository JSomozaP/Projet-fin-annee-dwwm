  providedIn: 'root'

export class CacheService {
  private memoryCache = new Map<string, CacheEntry>();
  private readonly TTL = 30000; // 30 secondes
  
  async get<T>(key: string, fetcher: () => Promise<T>): Promise<T> {
    // Level 1: Memory cache
    const cached = this.memoryCache.get(key);
    if (cached && Date.now() - cached.timestamp < this.TTL) {
      console.log(`üéØ Cache HIT (memory): ${key}`);
      return cached.data;
    }
    
    // Level 2: Database cache (impl√©ment√© c√¥t√© backend)
    try {
      const data = await fetcher();
      
      // Mise √† jour cache m√©moire
      this.memoryCache.set(key, {
        data,
        timestamp: Date.now()
      });
      
      console.log(`üì° Cache MISS: ${key} - Data fetched`);
      return data;
    } catch (error) {
      console.error(`‚ùå Cache error for ${key}:`, error);
      throw error;
    }
  }
  
  // Nettoyage automatique cache expir√©
  private cleanExpiredCache() {
    const now = Date.now();
    for (const [key, entry] of this.memoryCache.entries()) {
      if (now - entry.timestamp > this.TTL) {
        this.memoryCache.delete(key);
      }
    }
  }
}
